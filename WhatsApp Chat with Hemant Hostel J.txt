class Solution {
public:
    
    unordered_set<string> st;
    int maxe=0;
    unordered_map<int,vector<string>> map;

    vector<string> helper(string &s,int index){
        if(map.find(index)!=map.end()){
            return map[index];
        }
        vector<string> ans;
        if(index==s.size()){
            return {""};
        }
        for(int i=index;i<s.size() && i<index+maxe;i++){
            if(st.find(s.substr(index,i-index+1))!=st.end()){
                auto temp=helper(s,i+1);
                for(auto x:temp){
                    if(x==""){
                        ans.push_back(s.substr(index,i-index+1)+x);
                    }
                    else{
                        ans.push_back(s.substr(index,i-index+1)+" "+x);
                    }
                }
            }
        }
        return map[index]=ans;
    }
    
    vector<string> wordBreak(string s, vector<string>& dictionary) {
        st.clear();
        for(auto x:dictionary){
            st.insert(x);
            maxe=max(maxe,(int)x.size());
        }
        string curr="";
        return helper(s,0);
    }
};
